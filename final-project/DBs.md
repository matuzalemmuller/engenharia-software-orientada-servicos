### NewTec Database
```
CREATE SCHEMA `NewTec` ;
CREATE TABLE NewTec.PEDIDOS  (
	DIA					    DATETIME,
    CODIGO  			    INT NOT NULL AUTO_INCREMENT,
    CLIENTE 			    VARCHAR(40) NOT NULL,
    CPF 				    VARCHAR(15) NOT NULL,
    EMAIL 				    VARCHAR(40) NOT NULL,
    CEP 				    VARCHAR(10) NOT NULL,
    ESTADO 				    VARCHAR(2) NOT NULL,
    BAIRRO 				    VARCHAR(40) NOT NULL,
    CIDADE 				    VARCHAR(40) NOT NULL,
    RUA 				    VARCHAR(40) NOT NULL,
    NUMERO_CC 			    VARCHAR(20) NOT NULL,
    VALIDADE_CC 		    VARCHAR(10) NOT NULL,
    AUTORIZACAO_CC 		    VARCHAR(60) NOT NULL,
    QTD_REATOR_SOLAR 	    INTEGER NOT NULL,
    QTD_REATOR_ARK 		    INTEGER NOT NULL,
    QTD_ROBO_DOMESTICO 	    INTEGER NOT NULL,
    QTD_ROBO_SEGURANCA 	    INTEGER NOT NULL,
    QTD_ROBO_MEDICO 	    INTEGER NOT NULL,
    FRETE 				    INTEGER NOT NULL,
    TOTAL   			    REAL NOT NULL,
    TOTAL_COM_COMISSAO 	    REAL NOT NULL,
    PRIMARY KEY(CODIGO)
);
```

### BrRobotics Database

```
CREATE SCHEMA `BrRobotics` ;
CREATE TABLE BrRobotics.PEDIDOS  ( 
    CODIGO  				INT NOT NULL AUTO_INCREMENT,
    CLIENTE 				VARCHAR(40) NOT NULL,
    ESTADO 					VARCHAR(2) NOT NULL,
    ENDERECO				VARCHAR(40) NOT NULL,
    QTD_ROBO_DOMESTICO 		INTEGER NOT NULL,
    QTD_ROBO_SEGURANCA 		INTEGER NOT NULL,
    QTD_ROBO_MEDICO 		INTEGER NOT NULL,
    TOTAL_ROBO_DOMESTICO	REAL NOT NULL,
    TOTAL_ROBO_SEGURANCA	REAL NOT NULL,
	TOTAL_ROBO_MEDICO 		REAL NOT NULL,
    TOTAL   				REAL NOT NULL,
    PRIMARY KEY(CODIGO)
);
```
